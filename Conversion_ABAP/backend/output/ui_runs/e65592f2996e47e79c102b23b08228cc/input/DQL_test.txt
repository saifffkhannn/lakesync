REPORT z_sales_order_report.

SELECT
    vbak~vbeln,
    vbak~auart,
    vbak~erdat,
    kna1~name1,
    vbap~matnr,
    vbap~kwmeng,
    vbap~netwr
FROM vbak
INNER JOIN vbap
    ON vbak~vbeln = vbap~vbeln
INNER JOIN kna1
    ON vbak~kunnr = kna1~kunnr
INTO TABLE @DATA(lt_sales)
WHERE vbak~auart IN ('OR', 'TA')
  AND vbak~erdat >= '20240101'.

LOOP AT lt_sales INTO DATA(ls_sales).
  WRITE: /
      ls_sales-vbeln,
      ls_sales-name1,
      ls_sales-netwr.
ENDLOOP.
