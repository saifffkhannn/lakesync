@AbapCatalog.sqlViewName: 'ZVSALESANL'
@EndUserText.label: 'Sales Analytics'

define view Z_I_SalesAnalytics
  as select from vbak as Header

    inner join vbap as Item
      on Header.vbeln = Item.vbeln

{
    key Header.vbeln as SalesOrder,

    key Item.posnr as ItemNumber,

    Header.auart as OrderType,

    Item.netwr as NetValue,

    case
        when Item.netwr > 100000
             then 'HIGH'
        else 'NORMAL'
    end as SalesCategory

}
